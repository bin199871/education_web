## 4. 物理引擎详细设计

### 4.1 物理常量

```javascript
const PHYS = {
  m: 0.1,            // 质量 (kg)
  q: 5e-4,           // 电荷量 (C)
  E: 2000,           // 电场强度 (N/C)
  g: 10,             // 重力加速度 (m/s²)
  L: 1.0,            // 绳长 (m)

  // 导出量（预计算）
  get Fq() { return this.q * this.E; },     // F电 = 1.0 N
  get G()  { return this.m * this.g; },     // G = 1.0 N
  get F_result() {                          // F合 = √2 ≈ 1.41 N
    return Math.sqrt(this.Fq * this.Fq + this.G * this.G);
  },

  // 角度常量
  thetaTarget: 37,    // 目标角度 (°)
  sin37: 0.6,
  cos37: 0.8,
  deg: Math.PI / 180  // 角度转弧度系数
};
```

### 4.2 物理数据预计算

#### 数据一：第一次释放（θ: 0° → 90°）

```javascript
function computeSwing1(numSteps) {
  // 释放位置：θ=0° (A点正下方)
  // 摆动范围：0° → 90° (复合场中最大摆角)
  // 推导：mg·L(1-cosθ) = qE·L·sinθ → 1-cosθ = sinθ → θ_max = 90°
  const data = [];
  for (let i = 0; i <= numSteps; i++) {
    const theta_deg = 90 * (i / numSteps);
    const theta = theta_deg * PHYS.deg;
    const sinT = Math.sin(theta);
    const cosT = Math.cos(theta);

    // 几何量
    const h = PHYS.L * (1 - cosT);          // 高度差
    const x = PHYS.L * sinT;                // 水平位移

    // 能量
    const U = PHYS.G * PHYS.L * (1 - cosT) - PHYS.Fq * PHYS.L * sinT;
    const K = -U;  // 从θ=0°释放，U(0)=0
    const Ek = Math.max(K, 0);
    const Ep_g = PHYS.G * h;
    const Ep_e = -PHYS.Fq * x;

    // 速度
    const v = Math.sqrt(2 * Ek / PHYS.m);

    // 拉力 (沿绳方向)
    // T = mg·cosθ + qE·sinθ + mv²/L
    const T = PHYS.G * cosT + PHYS.Fq * sinT + PHYS.m * v * v / PHYS.L;

    data.push({ theta, theta_deg, h, x, v, Ek, Ep_g, Ep_e, Etot: Ek + Ep_g + Ep_e, T });
  }
  return data;
}
```

#### 数据二：第二次释放（θ: 90° → 0°）

```javascript
function computeSwing2(numSteps) {
  // 释放位置：θ=90° (水平位置)
  // 摆动范围：90° → 0° (最低点)
  const data = [];
  for (let i = 0; i <= numSteps; i++) {
    const theta_deg = 90 * (1 - i / numSteps);
    const theta = theta_deg * PHYS.deg;
    const sinT = Math.sin(theta);
    const cosT = Math.cos(theta);

    // 从90° 到 θ 的竖直位移和水平位移
    const h = PHYS.L * (1 - cosT);     // 相对最低点的高度
    // 从90°时高度为L，到θ时高度为L·cosθ→竖直位移 = L - L·cosθ = L(1-cosθ)
    // 从90°水平位移为L，到θ时水平位移为L·sinθ→水平位移 = L - L·sinθ
    
    // 能量：从θ=90°释放，势能零点在最低点
    // U(90°) = mg·L + 0 = 1.0J, U(0°) = 0
    // 从90°→0°, G做功 = mg·L = 1.0J (正功), F电做功 = qE·L = 1.0J (正功)
    // W总 = 2.0J
    const h_from_90 = PHYS.L * (1 - cosT);      // 从90°下降的竖直距离
    const x_from_90 = PHYS.L * (1 - sinT);       // 从90°到θ的水平位移
    const W_g = PHYS.G * h_from_90;
    const W_e = PHYS.Fq * x_from_90;
    const Ek = W_g + W_e;
    const v = Math.sqrt(2 * Ek / PHYS.m);

    const T = PHYS.G * cosT + PHYS.Fq * sinT + PHYS.m * v * v / PHYS.L;

    data.push({ theta, theta_deg, h, x: PHYS.L * sinT, v, Ek,
      Ep_g: PHYS.G * (PHYS.L - h_from_90),    // 相对最低点
      Ep_e: -PHYS.Fq * PHYS.L * sinT,
      Etot: Ek + PHYS.G * (PHYS.L - h_from_90) - PHYS.Fq * PHYS.L * sinT,
      T });
  }
  return data;
}
```

#### 数据三：θ=37° 附近的精细数据（用于高亮慢放）

```javascript
function computeHighlightData(fineSteps) {
  // 在 θ=34°~40° 区间精细插值
  const data = [];
  for (let i = 0; i <= fineSteps; i++) {
    const theta_deg = 34 + 6 * (i / fineSteps);
    const theta = theta_deg * PHYS.deg;
    const sinT = Math.sin(theta);
    const cosT = Math.cos(theta);

    // 同 computeSwing1 的计算方式
    const h = PHYS.L * (1 - cosT);
    const x = PHYS.L * sinT;
    const K = PHYS.Fq * PHYS.L * sinT - PHYS.G * PHYS.L * (1 - cosT);
    const Ek = Math.max(K, 0);
    const v = Math.sqrt(2 * Ek / PHYS.m);
    const T = PHYS.G * cosT + PHYS.Fq * sinT + PHYS.m * v * v / PHYS.L;

    data.push({ theta, theta_deg, h, x, v, Ek,
      Ep_g: PHYS.G * h,
      Ep_e: -PHYS.Fq * x,
      Etot: Ek + PHYS.G * h - PHYS.Fq * x,
      T });
  }
  return data;
}
```

### 4.3 物理状态查询接口

```javascript
const PhysicsEngine = {
  // 预计算数据
  swing1: null,           // 第一次释放 0°→90°
  swing2: null,           // 第二次释放 90°→0°
  highlight: null,        // 高亮区间 34°~40°
  numSteps: 120,          // 默认步数
  fineSteps: 60,          // 精细步数

  // 初始化：预计算所有数据
  init() {
    this.swing1 = computeSwing1(this.numSteps);
    this.swing2 = computeSwing2(this.numSteps);
    this.highlight = computeHighlightData(this.fineSteps);
  },

  // 根据场景和进度查询物理状态
  getState(sceneId, progress) {
    // progress: 0~1

    switch (sceneId) {
      case 'intro':
        // 静止在 θ=0°
        return this.swing1[0];

      case 'forces':
      case 'sol_q1':
        // 静止在 θ=0°，显示受力
        return this.swing1[0];

      case 'trans1':
        // 静止在 θ=0°
        return this.swing1[0];

      case 'swing_q2':
        // 第一次释放：0° → 90°
        return this._interpolate(this.swing1, progress);

      case 'highlight':
        // 37°附近精细区间
        return this._interpolate(this.highlight, progress);

      case 'sol_q2':
        // 停在 θ=37°
        return this._interpolate(this.swing1, 37 / 90);

      case 'trans2':
        // 停在 θ=0°
        return this.swing1[0];

      case 'pull_90':
        // 拉球过程：θ=0° → 90°，无物理
        return { theta: progress * Math.PI / 2, theta_deg: progress * 90,
          h: 0, x: 0, v: 0, Ek: 0, Ep_g: 0, Ep_e: 0, Etot: 0, T: 0 };

      case 'swing_q3':
        // 第二次释放：90° → 0°
        return this._interpolate(this.swing2, progress);

      case 'sol_q3':
        // 停在 θ=0° (最低点)
        return this._interpolate(this.swing2, 1);

      case 'ending':
        // 停在 θ=0°
        return this._interpolate(this.swing2, 1);

      default:
        return this.swing1[0];
    }
  },

  // 线性插值
  _interpolate(data, progress) {
    const idx = progress * (data.length - 1);
    const i = Math.floor(idx);
    const frac = idx - i;
    if (i >= data.length - 1) return data[data.length - 1];
    if (i < 0) return data[0];

    // 对数值字段插值
    const a = data[i], b = data[i + 1];
    return {
      theta: a.theta + (b.theta - a.theta) * frac,
      theta_deg: a.theta_deg + (b.theta_deg - a.theta_deg) * frac,
      h: a.h + (b.h - a.h) * frac,
      x: a.x + (b.x - a.x) * frac,
      v: a.v + (b.v - a.v) * frac,
      Ek: a.Ek + (b.Ek - a.Ek) * frac,
      Ep_g: a.Ep_g + (b.Ep_g - a.Ep_g) * frac,
      Ep_e: a.Ep_e + (b.Ep_e - a.Ep_e) * frac,
      Etot: a.Etot + (b.Etot - a.Etot) * frac,
      T: a.T + (b.T - a.T) * frac
    };
  }
};
```

### 5.1 核心实现

```javascript
const FrameController = {
  // 状态
  frame: 0,             // 当前帧 (float，支持子帧精度)
  totalFrames: 10800,   // 总帧数 (3min × 60fps)
  fps: 60,
  isPlaying: false,
  speed: 1.0,
  isComplete: false,

  // 内部
  _animId: null,
  _lastTimestamp: 0,
  _slowMotionOverride: null,   // { startFrame, endFrame, speed }
  _callbacks: {
    onFrame: null,       // function(frame, sceneId, physics)
    onComplete: null,
    onPause: null,
    onResume: null,
    onSeek: null
  },

  // 帧率时间常量
  get frameDuration() { return 1000 / this.fps; },  // 16.67ms

  // 方法
  init(totalFrames, fps) {
    this.totalFrames = totalFrames || 10800;
    this.fps = fps || 60;
  },

  start() {
    this._lastTimestamp = 0;
    this._loop();
  },

  _loop(timestamp) {
    if (!this._lastTimestamp) this._lastTimestamp = timestamp;
    const delta = timestamp - this._lastTimestamp;
    this._lastTimestamp = timestamp;

    if (this.isPlaying) {
      // 计算有效速度（考虑慢放）
      const effectiveSpeed = this._getEffectiveSpeed();

      // 累加帧数
      const frameDelta = (delta / 1000) * this.fps * effectiveSpeed;
      this.frame = Math.min(this.frame + frameDelta, this.totalFrames - 1);

      // 回调
      this._notifyFrame();

      // 检查是否播完
      if (this.frame >= this.totalFrames - 1) {
        this.isPlaying = false;
        this.isComplete = true;
        if (this._callbacks.onComplete) this._callbacks.onComplete();
      }
    }

    this._animId = requestAnimationFrame((t) => this._loop(t));
  },

  _getEffectiveSpeed() {
    let s = this.speed;
    if (this._slowMotionOverride) {
      const { startFrame, endFrame, slowSpeed } = this._slowMotionOverride;
      if (this.frame >= startFrame && this.frame < endFrame) {
        s *= slowSpeed;
      }
    }
    return s;
  },

  _notifyFrame() {
    const frameInt = Math.floor(this.frame);
    if (this._callbacks.onFrame) {
      this._callbacks.onFrame(frameInt);
    }
  },

  play() {
    this.isPlaying = true;
    this.isComplete = false;
    this._lastTimestamp = 0;
    if (this._callbacks.onResume) this._callbacks.onResume();
  },

  pause() {
    this.isPlaying = false;
    if (this._callbacks.onPause) this._callbacks.onPause();
  },

  toggle() {
    if (this.isComplete) {
      this.seek(0);
      this.play();
    } else if (this.isPlaying) {
      this.pause();
    } else {
      this.play();
    }
  },

  seek(targetFrame) {
    this.frame = Math.max(0, Math.min(targetFrame, this.totalFrames - 1));
    this.isComplete = false;
    if (this._callbacks.onSeek) this._callbacks.onSeek(Math.floor(this.frame));
    this._notifyFrame();
  },

  setSpeed(speed) {
    this.speed = speed;
  },

  reset() {
    this.frame = 0;
    this.isComplete = false;
    this.isPlaying = false;
    this._slowMotionOverride = null;
    this._notifyFrame();
  },

  // 注册慢放区间
  setSlowMotion(startFrame, endFrame, slowSpeed) {
    this._slowMotionOverride = { startFrame, endFrame, slowSpeed: slowSpeed || 0.3 };
  },

  clearSlowMotion() {
    this._slowMotionOverride = null;
  }
};
```

### 6.3 核心逻辑

```javascript
const SceneDispatcher = {
  _currentSceneId: null,

  getScene(frame) {
    const pct = frame / totalFrames;
    for (const s of SCENE_DEFS) {
      if (pct >= s.startPct && pct < s.endPct) return s.id;
    }
    return 'ending';
  },

  getSceneProgress(frame, sceneId) {
    const s = SCENE_DEFS.find(x => x.id === sceneId);
    if (!s) return 0;
    const range = s.endPct - s.startPct;
    if (range <= 0) return 1;
    const pct = frame / totalFrames;
    return Math.min(1, Math.max(0, (pct - s.startPct) / range));
  },

  getSceneDef(sceneId) {
    return SCENE_DEFS.find(x => x.id === sceneId);
  },

  onFrame(frame) {
    const newScene = this.getScene(frame);
    const changed = newScene !== this._currentSceneId;
    this._currentSceneId = newScene;
    return { sceneId: newScene, changed };
  },

  // 场景进入/退出的hook点
  _enterCallbacks: {},
  _exitCallbacks: {},

  onEnter(sceneId, callback) {
    if (!this._enterCallbacks[sceneId]) this._enterCallbacks[sceneId] = [];
    this._enterCallbacks[sceneId].push(callback);
  },

  onExit(sceneId, callback) {
    if (!this._exitCallbacks[sceneId]) this._exitCallbacks[sceneId] = [];
    this._exitCallbacks[sceneId].push(callback);
  },

  fireEnter(sceneId) {
    (this._enterCallbacks[sceneId] || []).forEach(cb => cb());
  },

  fireExit(sceneId) {
    (this._exitCallbacks[sceneId] || []).forEach(cb => cb());
  }
};
```

```javascript
const PanelController = {
  panelEl: null,
  currentSceneId: null,
  currentProgress: 0,

  init() {
    this.panelEl = document.getElementById('solutionPanel');
    this._buildPanels();
  },

  _buildPanels() {
    // 根据 PANEL_CONTENTS 生成 DOM
    for (const [sceneId, content] of Object.entries(PANEL_CONTENTS)) {
      const div = document.createElement('div');
      div.id = 'panel-' + sceneId;
      div.className = 'panel-content';
      div.style.display = 'none';

      // 标题
      const title = document.createElement('div');
      title.className = 'panel-title';
      title.style.color = content.color;
      title.textContent = content.title;
      div.appendChild(title);

      // 步骤
      content.steps.forEach((step, i) => {
        const row = document.createElement('div');
        row.className = 'step-row';
        row.id = sceneId + '-step-' + i;

        const badge = document.createElement('span');
        badge.className = 'step-badge';
        badge.textContent = '①③②④⑤⑥⑦⑧⑨'[i] || (i + 1);
        row.appendChild(badge);

        const text = document.createElement('span');
        text.innerHTML = step.html;
        row.appendChild(text);

        div.appendChild(row);
      });

      // 答案框
      const answerBox = document.createElement('div');
      answerBox.className = 'answer-box';
      answerBox.id = sceneId + '-answer';
      const answerText = document.createElement('span');
      answerText.className = 'answer-text';
      answerText.innerHTML = content.answer;
      answerBox.appendChild(answerText);
      div.appendChild(answerBox);

      this.panelEl.appendChild(div);
    }
  },

  update(sceneId, progress) {
    // 检查是否需要切换面板内容
    if (sceneId !== this.currentSceneId) {
      // 隐藏旧面板
      if (this.currentSceneId) {
        const oldPanel = document.getElementById('panel-' + this.currentSceneId);
        if (oldPanel) oldPanel.style.display = 'none';
      }
      // 显示新面板
      const newPanel = document.getElementById('panel-' + sceneId);
      if (newPanel) {
        newPanel.style.display = 'block';
        // 滑入动画
        this.panelEl.classList.add('visible');
      } else {
        // 非解题场景：隐藏面板
        this.panelEl.classList.remove('visible');
      }
      this.currentSceneId = sceneId;
      this.currentProgress = 0;
    }

    if (!PANEL_CONTENTS[sceneId]) return;

    // 更新步骤可见性
    const steps = PANEL_CONTENTS[sceneId].steps;
    const visibleCount = Math.min(
      Math.floor(progress * steps.length / 0.85),
      steps.length
    );

    for (let i = 0; i < steps.length; i++) {
      const el = document.getElementById(sceneId + '-step-' + i);
      if (el) {
        el.classList.toggle('visible', i < visibleCount);
      }
    }

    // 答案框：最后15%进度中淡入
    const answerEl = document.getElementById(sceneId + '-answer');
    if (answerEl) {
      const answerProgress = (progress - 0.85) / 0.15;
      answerEl.classList.toggle('visible', answerProgress > 0);
    }

    this.currentProgress = progress;
  },

  hide() {
    this.panelEl.classList.remove('visible');
    this.currentSceneId = null;
  }
};
```

```javascript
const UIController = {
  playBtn: null,

  progressBar: null,
  progressFill: null,

  init() {

      fc.toggle();
      this._updatePlayButton();
    });

    this.rewindBtn.addEventListener('click', () => {
      fc.reset();
      this._syncAll();
    });

    this.progressBar.addEventListener('click', (e) => {
      const rect = this.progressBar.getBoundingClientRect();
      const pct = Math.max(0, Math.min((e.clientX - rect.left) / rect.width, 1));
      fc.seek(Math.floor(pct * fc.totalFrames));
      this._syncAll();
    });

    this.speedBtn.addEventListener('click', () => {
      const speeds = [0.5, 1.0, 1.5, 2.0];
      const currentIdx = speeds.indexOf(fc.speed);
      const nextIdx = (currentIdx + 1) % speeds.length;
      fc.setSpeed(speeds[nextIdx]);
      this.speedBtn.textContent = speeds[nextIdx] + '×';
    });

    // 键盘
    document.addEventListener('keydown', (e) => {
      switch (e.code) {
        case 'Space':
          e.preventDefault();
          fc.toggle();
          this._updatePlayButton();
          break;
        case 'ArrowRight':
          fc.seek(fc.frame + 5 * fc.fps);
          this._syncAll();
          break;
        case 'ArrowLeft':
          fc.seek(fc.frame - 5 * fc.fps);
          this._syncAll();
          break;
        case 'KeyR':
          fc.reset();
          this._syncAll();
          break;
      }
    });
  },

  _updatePlayButton() {
    const fc = FrameController;
    if (fc.isComplete) {
      this.playBtn.textContent = '↺';
    } else if (fc.isPlaying) {
      this.playBtn.textContent = '⏸';
    } else {
      this.playBtn.textContent = '▶';
    }
  },

  // 进度/时间更新
  updateProgress(frame, total) {
    const pct = total > 0 ? (frame / total) * 100 : 0;
    this.progressFill.style.width = Math.min(pct, 100) + '%';

    const curSec = Math.floor(frame / 60);
    const totalSec = Math.floor(total / 60);
    this.timeLabel.textContent =
      String(Math.floor(curSec / 60)) + ':' + String(curSec % 60).padStart(2, '0')
      + ' / ' +
      String(Math.floor(totalSec / 60)) + ':' + String(totalSec % 60).padStart(2, '0');
  },

  _syncAll() {
    const fc = FrameController;
    this._updatePlayButton();
    this.updateProgress(fc.frame, fc.totalFrames);
  }
};
```

### 12.2 完整帧回调

```javascript
function onFrame(frame) {
  const { sceneId, changed } = SceneDispatcher.onFrame(frame);
  const progress = SceneDispatcher.getSceneProgress(frame, sceneId);
  const physics = PhysicsEngine.getState(sceneId, progress);

  // 场景切换事件
  if (changed) {
    SceneTitleController.update(sceneId);

    // 慢放控制
    if (sceneId === 'highlight') {
      const s = SCENE_DEFS.find(x => x.id === 'highlight');
      FrameController.setSlowMotion(
        Math.floor(s.startPct * totalFrames),
        Math.floor(s.endPct * totalFrames),
        0.3
      );
    } else {
      FrameController.clearSlowMotion();
    }

    // 场景进入时重置解题面板进度
    PanelController.update(sceneId, 0);
  }

  // 渲染
  CanvasRenderer.render(sceneId, physics);

  // 解题面板
  if (['sol_q1', 'sol_q2', 'sol_q3'].includes(sceneId)) {
    PanelController.update(sceneId, progress);
  } else {
    PanelController.hideVisible();
  }

  // 答案汇总
  EndingController.update(sceneId, progress);

  // 过渡特效
  if (['trans1', 'trans2'].includes(sceneId)) {
    CanvasRenderer._drawTransitionOverlay(sceneId, progress);
  }

  // 拉球
  if (sceneId === 'pull_90') {
    CanvasRenderer._drawPullAnimation(null, null, physics.theta, progress);
  }

  // 更新UI
  UIController.updateProgress(frame, totalFrames);
}

// 注册到帧控制器
FrameController._callbacks.onFrame = onFrame;
FrameController._callbacks.onComplete = () => {
  UIController._updatePlayButton();
};
```
