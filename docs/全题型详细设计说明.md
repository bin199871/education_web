#### 2. 物理引擎

```javascript
function getCircularState(theta) {
  // theta: 0~2π 表示小球在轨道上的角度
  // 0=最低点, π=最高点
  const R = P.R;
  const v_top = P.v1;    // 最高点速度
  const v_bot = P.v2;    // 最低点速度
  const g = P.g;
  const m = P.m;

  // 任意角度的速度（能量守恒近似）
  const h = R * (1 - Math.cos(theta));  // 相对最低点高度
  const v = Math.sqrt(v_bot*v_bot - 2*g*h);

  // 向心力
  const Fc = m * v*v / R;
  const N = Fc + m*g*Math.cos(theta);  // 轨道支持力

  // 笛卡尔坐标
  const cx = 480, cy = 300;  // 轨道圆心
  const px = cx + R*300 * Math.sin(theta);
  const py = cy - R*300 * Math.cos(theta);

  return {theta, v, N, Fc, px, py, h};
}
```

```javascript
// 简谐振动参数
const omega = Math.sqrt(P.k / P.m);
const T = 2 * Math.PI / omega;
const v_max = P.A * omega;

function getSpringState(prog) {
  // prog: 0~1 对应一个完整周期
  const t = prog * T;
  const x = P.A * Math.cos(omega * t);         // 位移
  const v = -P.A * omega * Math.sin(omega * t); // 速度
  const a = -P.A * omega * omega * Math.cos(omega * t); // 加速度
  const Ek = 0.5 * P.m * v * v;
  const Ep = 0.5 * P.k * x * x;
  const E_total = 0.5 * P.k * P.A * P.A;
  return {t, x, v, a, Ek, Ep, E_total};
}
```

```javascript
function drawSpring(ctx, x1, y1, x2, y2, coils) {
  ctx.beginPath();

  ctx.stroke();
```

#### 2. 物理引擎

```javascript
function getConveyorState(prog) {
  // 先计算关键物理量
  const a = P.mu * P.g;
  const t_co = P.v0 / a;
  const x_co = P.v0 * P.v0 / (2 * a);
  const t_total = t_co * 1.5; // 总模拟时间

  const t = prog * t_total;
  let v_block, x_block;

  if (t <= t_co) {
    // 加速阶段
    v_block = a * t;
    x_block = 0.5 * a * t * t;
  } else {
    // 匀速阶段
    v_block = P.v0;
    x_block = x_co + P.v0 * (t - t_co);
  }

  const Q = P.mu * P.m * P.g * (x_block - 0.5 * a * Math.min(t, t_co) * Math.min(t, t_co));
  const Ek = 0.5 * P.m * v_block * v_block;

  return {t, v_block, x_block, a, Q, Ek, t_co, x_co};
}
```

```javascript
const v_max = P.P / P.f;  // 最大速度

function getLocomotiveState(prog) {
  const sim_time = 20; // 模拟20秒
  const t = prog * sim_time;
  let v, a;
  const steps = 1000;
  const dt = sim_time / steps;
  v = 0;
  for (let i = 0; i <= steps; i++) {
    const time = i * dt;
    if (time > t) break;
    const F = v < v_max ? P.P / Math.max(v, 0.01) : P.f;
    a = (F - P.f) / P.m;
    v += a * dt;
  }
  const Ek = 0.5 * P.m * v * v;
  const W = P.P * t; // 牵引力做功（近似）
  return {t, v, a, Ek, W, v_max};
}
```

#### 2. 物理引擎

```javascript
// 弹性碰撞结果
const v1_after = ((P.m1 - P.m2) * P.v1 + 2 * P.m2 * P.v2) / (P.m1 + P.m2);
const v2_after = (2 * P.m1 * P.v1 + (P.m2 - P.m1) * P.v2) / (P.m1 + P.m2);

function getCollisionState(prog) {
  // 三阶段：碰撞前 → 碰撞 → 碰撞后
  let x1, x2, v1, v2;
  if (prog < 0.3) {
    // 碰撞前
    const p = prog / 0.3;
    x1 = 200 - p * 80;
    x2 = 300 - p * 20;
    v1 = P.v1; v2 = 0;
  } else if (prog < 0.4) {
    // 碰撞瞬间（重叠）
    x1 = 250; x2 = 250;
    v1 = 0; v2 = 0;
  } else {
    // 碰撞后
    const p = (prog - 0.4) / 0.6;
    x1 = 250 - p * 100;
    x2 = 250 + p * 150;
    v1 = v1_after; v2 = v2_after;
  }
  const p1 = P.m1 * v1, p2 = P.m2 * v2;
  const Ek1_before = 0.5 * P.m1 * P.v1 * P.v1;
  const Ek_after = 0.5 * P.m1 * v1_after * v1_after + 0.5 * P.m2 * v2_after * v2_after;
  const loss = Ek1_before - Ek_after;
  return {x1, x2, v1, v2, p1, p2, loss, elastic: P.elastic};
}
```

#### 2. 物理引擎

```javascript
const v_common = P.m * P.v0 / (P.m + P.M);
const a1 = -P.mu * P.g;
const a2 = P.mu * P.m * P.g / P.M;
const t_co = P.v0 / (P.g * P.mu * (1 + P.m / P.M));
const x1 = P.v0 * t_co + 0.5 * a1 * t_co * t_co;
const x2 = 0.5 * a2 * t_co * t_co;
const dx = x1 - x2;

function getBoardState(prog) {
  const t = prog * t_co * 1.5;
  if (t <= t_co) {
    v1 = P.v0 + a1 * t;
    v2 = a2 * t;
    x1 = P.v0 * t + 0.5 * a1 * t * t;
    x2 = 0.5 * a2 * t * t;
  } else {
    v1 = v_common; v2 = v_common;
    x1 = x1 + v_common * (t - t_co);
```
