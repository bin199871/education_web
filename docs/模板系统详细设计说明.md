# ================================================================
#  2. TemplateEngine 类
# ================================================================

class TemplateEngine:
    """模板引擎：匹配 → 提取 → 渲染"""

    def __init__(self, templates_dir: str = None, llm_config: dict = None):
        self.templates_dir = Path(templates_dir or (
            Path(__file__).parent / 'templates'
        ))
        self.templates_dir.mkdir(parents=True, exist_ok=True)
        self.llm_config = llm_config or {}
        self.registry = dict(TEMPLATE_REGISTRY)  # 可运行时覆盖

    # -----------------------------------------------------------
    #  2.1 模板列表
    # -----------------------------------------------------------

    def list_templates(self) -> list:
        """返回所有可用模板（排除 param_schema 内部结构）"""
        result = []
        for tid, tpl in self.registry.items():
            result.append({
                'id': tid,
                'label': tpl['label'],
                'params': list(tpl['params']),
                'tags': list(tpl['tags']),
                'param_info': {
                    k: {'type': v['type'], 'unit': v.get('unit', ''), 'default': v['default']}
                    for k, v in tpl['param_schema'].items()
                }
            })
        return result

    # -----------------------------------------------------------
    #  2.2 模板匹配算法
    # -----------------------------------------------------------

    def match_template(self, analysis: dict) -> Optional[str]:
        """
        根据 Layer1 分析结果匹配最佳模板。
        
        输入格式：
        {
            'topic': '电场',
            'tags': ['电场', '圆周运动', '单摆'],
            'concepts': ['电场力', '动能定理', ...],
            'text': '在竖直平面内存在水平向右的匀强电场...'
        }
        
        返回：模板ID，无匹配返回 None
        """
        topic = analysis.get('topic', '') or ''
        tags = analysis.get('tags', []) or []
        concepts = analysis.get('concepts', []) or []
        text = analysis.get('text', '') or ''

        # 构建评分空间
        scores: List[Tuple[str, int, List[str]]] = []  # (tid, score, matched_tags)

        for tid, tpl in self.registry.items():
            score = 0
            matched = []

            # ---- 维度1: 标签精确匹配（权重最高） ----
            for tag in tpl['tags']:
                if tag in tags:
                    score += 10
                    matched.append(f'标签精确:{tag}')
                elif tag in topic:
                    score += 8
                    matched.append(f'主题匹配:{tag}')

            # ---- 维度2: 关键词模糊匹配 ----
            search_text = (topic + ' ' + ' '.join(tags) + ' ' +
                          ' '.join(concepts) + ' ' + text).lower()
            for tag in tpl['tags']:
                if tag.lower() in search_text:
                    score += 3
                    matched.append(f'关键词:{tag}')

            # ---- 维度3: 题型关联匹配 ----
            # 如 "电场+单摆" 同时出现时增强 electric_pendulum 的分数
            tid_synergies = {
                'electric_pendulum': ['电场', '单摆', '复合场'],
                'inclined_plane': ['斜面', '摩擦', '粗糙'],
                'projectile': ['平抛', '抛体', '水平初速度']
            }
            if tid in tid_synergies:
                synergy_hits = sum(
                    1 for kw in tid_synergies[tid] if kw in search_text
                )
                if synergy_hits >= 2:
                    bonus = synergy_hits * 5
                    score += bonus
                    matched.append(f'关联增强:+{bonus}')

            if score > 0:
                scores.append((tid, score, matched))

        if not scores:
            return None

        # 排序：先按分数降序，同分按模板在注册表中的顺序
        registry_order = {tid: i for i, tid in enumerate(self.registry.keys())}
        scores.sort(key=lambda x: (-x[1], registry_order.get(x[0], 999)))

        best_tid = scores[0][0]
        logger.info(f'模板匹配结果: {best_tid} (得分:{scores[0][1]}, '
                   f'匹配项:{scores[0][2]})')

        return best_tid

    # -----------------------------------------------------------
    #  2.3 LLM 参数提取
    # -----------------------------------------------------------

    def extract_params(self, problem_text: str, template_id: str) -> dict:
        """
        调用 LLM 从题目文本中提取模板所需参数。
        
        输入：
            problem_text: 完整的题目文本
            template_id: 模板ID
        
        返回：
            填充了所有必需参数的 dict
        """
        tpl = self.registry.get(template_id)
        if not tpl:
            raise ValueError(f'未知模板: {template_id}')

        # 构建 prompt
        prompt = self._build_extraction_prompt(problem_text, tpl)

        # 调用 LLM
        llm_response = self._call_llm(prompt)

        # 解析 JSON
        params = self._parse_llm_json(llm_response)

        # 校验和补充
        params = self._validate_and_fill(params, tpl)

        # 注入题目文本和答案相关的辅助字段
        params['problem_text'] = problem_text
        params['title'] = tpl['label']

        return params

    def _build_extraction_prompt(self, problem_text: str, tpl: dict) -> str:
        """构建 LLM 参数提取 prompt"""
        
        # 构建参数描述
        schema_lines = []
        for param_name, info in tpl['param_schema'].items():
            desc = (f'"{param_name}": float, '
                   f'单位 {info.get("unit", "-")}, '
                   f'默认值 {info["default"]}, '
                   f'范围 [{info.get("min", "-")}, {info.get("max", "-")}]')
            schema_lines.append(f'  {desc}')

        schema_str = '\n'.join(schema_lines)

        # 构建题型特定的示例
        examples = {
            'electric_pendulum': '''
示例输出：
{
  "m": 0.1, "q": 5e-4, "E": 2000, "L": 1.0, "g": 10, "theta_target": 37,
  "question1": "求电场力的大小F电和合力大小F合",
  "question2": "求θ=37°时的速度大小v",
  "question3": "求细线拉力的最大值Tmax",
  "answer1": "F电=1.0N, F合=√2≈1.41N",
  "answer2": "v=2√2≈2.83m/s",
  "answer3": "Tmax=5.0N"
}''',
            'inclined_plane': '''
示例输出：
{
  "m": 2.0, "theta": 30, "L": 4.0, "mu": 0.2, "g": 10,
  "question1": "求物体滑到底端时的速度",
  "question2": "求摩擦力做的功",
  "question3": "求粗糙面滑行距离",
  "answer1": "v=6.32m/s",
  "answer2": "Wf=-13.86J",
  "answer3": "s=10.2m"
}''',
            'projectile': '''
示例输出：
{
  "v0": 10, "h": 5, "g": 10,
  "question1": "求落地时间",
  "question2": "求水平射程",
  "question3": "求落地速度大小和方向",
  "answer1": "t=1s",
  "answer2": "x=10m",
  "answer3": "v=10√2≈14.1m/s, θ=45°"
}'''
        }

        example = examples.get(template_id, '')

        prompt = f"""你是一个高中物理题参数提取器。请从以下题目中提取物理参数，输出为纯JSON格式。

题目文本：
{problem_text}

题型：{tpl['label']}

需要提取的参数：
{schema_str}

额外提取（从题目中识别）：
  "question1": "第(1)问的文字描述"
  "question2": "第(2)问的文字描述"
  "question3": "第(3)问的文字描述"
  "answer1": "第(1)问的答案（含计算过程和结果）"
  "answer2": "第(2)问的答案（含计算过程和结果）"
  "answer3": "第(3)问的答案（含计算过程和结果）"

{example if example else ''}

输出规则：
1. 数值用 float 类型（如 5e-4, 2000.0）
2. 角度用 度（°）为单位，不要转弧度
3. 不包含单位字符（如不写 "10kg" 而写 10.0）
4. 题目未明确给出的参数用默认值
5. 答案保持与题目中相同的有效数字位数
6. 只输出 JSON，不要任何额外文字、解释、或代码块标记"""

        return prompt

    def _call_llm(self, prompt: str) -> str:
        """
        调用 LLM API。
        复用层1的 LLM 基础设施。
        """
        # 从配置中读取 LLM 设置
        api_key = self.llm_config.get('api_key') or os.getenv('LLM_API_KEY', '')
        api_url = self.llm_config.get('api_url') or os.getenv('LLM_API_URL',
            'https://api.deepseek.com/v1/chat/completions')
        model = self.llm_config.get('model') or os.getenv('LLM_MODEL',
            'deepseek-v4-flash')

        headers = {
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        }

        payload = {
            'model': model,
            'messages': [
                {
                    'role': 'system',
                    'content': '你是一个物理题参数提取助手。你只输出纯JSON，没有任何额外文字。'
                },
                {
                    'role': 'user',
                    'content': prompt
                }
            ],
            'temperature': 0.1,   # 低温度保证确定性输出
            'max_tokens': 2000,
            'response_format': {'type': 'json_object'}  # 如果API支持
        }

        try:
            import httpx
            with httpx.Client(timeout=30.0) as client:
                resp = client.post(api_url, json=payload, headers=headers)
                resp.raise_for_status()
                data = resp.json()
                return data['choices'][0]['message']['content']
        except ImportError:
            # fallback to urllib
            import urllib.request
            import urllib.error
            body = json.dumps(payload).encode('utf-8')
            req = urllib.request.Request(api_url, data=body, headers=headers,
                                        method='POST')
            try:
                with urllib.request.urlopen(req, timeout=30) as resp:
                    data = json.loads(resp.read().decode('utf-8'))
                    return data['choices'][0]['message']['content']
            except Exception as e:
                logger.error(f'LLM call failed: {e}')
                raise
        except Exception as e:
            logger.error(f'LLM call failed: {e}')
            raise

    def _parse_llm_json(self, response: str) -> dict:
        """
        解析 LLM 返回的 JSON。
        处理可能的 markdown 代码块包裹、前后空格等。
        """
        if not response or not response.strip():
            logger.warning('LLM 返回空响应，使用默认参数')
            return {}

        text = response.strip()

        # 尝试去掉 markdown 代码块标记
        if text.startswith('```'):
            text = re.sub(r'^```(?:json)?\s*', '', text)
            text = re.sub(r'\s*```$', '', text)

        # 尝试提取第一个 { } 块
        brace_start = text.find('{')
        brace_end = text.rfind('}')
        if brace_start >= 0 and brace_end > brace_start:
            text = text[brace_start:brace_end + 1]

        try:
            params = json.loads(text)
            if not isinstance(params, dict):
                logger.warning(f'LLM 返回非字典JSON: {type(params)}')
                return {}
            return params
        except json.JSONDecodeError as e:
            logger.error(f'LLM JSON 解析失败: {e}\n响应内容: {text[:500]}')
            return {}

    def _validate_and_fill(self, params: dict, tpl: dict) -> dict:
        """
        校验和补充参数。
        - 缺失参数 → 默认值
        - 类型转换 → float
        - 范围校验 → 超范围则用默认值
        """
        result = dict(params)

        for key, info in tpl['param_schema'].items():
            if key not in result or result[key] is None:
                # 缺失 → 默认值
                result[key] = info['default']
                logger.info(f'参数 {key} 缺失，使用默认值 {info["default"]}')
            else:
                try:
                    val = float(result[key])
                    # 范围校验
                    if 'min' in info and val < info['min']:
                        logger.warning(f'参数 {key}={val} < 最小值 {info["min"]}，重置为默认值')
                        val = info['default']
                    if 'max' in info and val > info['max']:
                        logger.warning(f'参数 {key}={val} > 最大值 {info["max"]}，重置为默认值')
                        val = info['default']
                    result[key] = val
                except (ValueError, TypeError):
                    logger.warning(f'参数 {key}={result[key]} 类型错误，使用默认值 {info["default"]}')
                    result[key] = info['default']

        # 确保 questions/answers 存在
        for i in range(1, 4):
            q_key = f'question{i}'
            a_key = f'answer{i}'
            if q_key not in result or not result[q_key]:
                result[q_key] = f'第({i})问'
            if a_key not in result or not result[a_key]:
                result[a_key] = ''

        return result

    # -----------------------------------------------------------
    #  2.4 参数校验（独立接口）
    # -----------------------------------------------------------

    def validate_params(self, template_id: str, params: dict) -> list:
        """
        校验参数是否符合模板 schema。
        返回错误信息列表，空列表表示全部通过。
        """
        errors = []
        tpl = self.registry.get(template_id)
        if not tpl:
            return [f'未知模板: {template_id}']

        for key, info in tpl['param_schema'].items():
            if key not in params or params[key] is None:
                errors.append(f'缺少必需参数: {key}')
                continue

            try:
                val = float(params[key])
                if 'min' in info and val < info['min']:
                    errors.append(f'{key}={val} < 最小值 {info["min"]}')
                if 'max' in info and val > info['max']:
                    errors.append(f'{key}={val} > 最大值 {info["max"]}')
            except (ValueError, TypeError):
                errors.append(f'{key} 不是有效的数值: {params[key]}')

        return errors

    # -----------------------------------------------------------
    #  2.5 模板渲染
    # -----------------------------------------------------------

    def render(self, template_id: str, params: dict) -> str:
        """
        读取模板 HTML → 注入参数 → 返回完整 HTML。
        
        注入方式：
        1. TEMPLATE_PARAMS = null → TEMPLATE_PARAMS = {json}
        2. [[TITLE]] → 题目标题
        """
        tpl = self.registry.get(template_id)
        if not tpl:
            raise ValueError(f'未知模板: {template_id}')

        template_path = self.templates_dir / tpl['file']
        if not template_path.exists():
            raise FileNotFoundError(
                f'模板文件不存在: {template_path}\n'
                f'请确保模板文件位于 {self.templates_dir} 目录下'
            )

        with open(template_path, 'r', encoding='utf-8') as f:
            html = f.read()

        # ---- 注入参数 ----
        # 提取 questions/answers 为独立字段，方便模板使用
        inject_params = dict(params)
        inject_params['questions'] = [
            {
                'title': params.get(f'question{i}', f'第({i})问'),
                'answer': params.get(f'answer{i}', '')
            }
            for i in range(1, 4)
        ]

        param_json = json.dumps(inject_params, ensure_ascii=False, indent=2)
        html = html.replace(
            'var TEMPLATE_PARAMS = null;',
            f'var TEMPLATE_PARAMS = {param_json};'
        )

        # ---- 注入标题 ----
        title = params.get('title', tpl['label'])
        html = html.replace('[[TITLE]]', title)

        # ---- 注入生成标记 ----
        gen_time = datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        html = html.replace('[[GENERATED_TIME]]', gen_time)

        return html

    # -----------------------------------------------------------
    #  2.6 热加载支持
    # -----------------------------------------------------------

    def reload_registry(self):
        """重新加载注册表（用于新增模板后无需重启服务）"""
        self.registry = dict(TEMPLATE_REGISTRY)
        logger.info(f'模板注册表已重新加载，共 {len(self.registry)} 个模板')


# ================================================================
#  3. 便捷工厂函数
# ================================================================

_engine_instance = None

def get_engine(llm_config: dict = None) -> TemplateEngine:
    """获取全局单例引擎实例"""
    global _engine_instance
    if _engine_instance is None:
        _engine_instance = TemplateEngine(llm_config=llm_config)
    return _engine_instance
```

## 2. API 端点详细设计

### 2.1 Pydantic 模型

```python
# server.py 中新增

from pydantic import BaseModel
from typing import Optional, List, Dict, Any

# ---- 请求模型 ----

class TemplateGenerateRequest(BaseModel):
    """生成模板动画请求"""
    text: str                              # 题目文本
    template_id: Optional[str] = None      # 指定模板（null则自动匹配）

class TemplatePreviewRequest(BaseModel):
    """预览模板请求"""
    template_id: str                       # 模板ID
    params: Dict[str, Any]                 # 参数

# ---- 响应模型 ----

class TemplateInfo(BaseModel):
    """模板信息"""
    id: str
    label: str
    params: List[str]
    tags: List[str]
    param_info: Dict[str, Any]

class TemplateListResponse(BaseModel):
    """模板列表响应"""
    templates: List[TemplateInfo]

class TemplateGenerateResponse(BaseModel):
    """生成模板动画响应"""
    success: bool
    template_id: Optional[str] = None
    params: Optional[Dict[str, Any]] = None
    html: Optional[str] = None
    fallback: bool = False
    message: str = ''

class TemplateErrorResponse(BaseModel):
    """错误响应"""
    success: bool = False
    error: str
```

### 2.2 端点实现

```python
# server.py 中新增

from fastapi import APIRouter, HTTPException
from fastapi.responses import HTMLResponse
from template_engine import get_engine

router = APIRouter(prefix='/api/template', tags=['模板动画'])


@router.get('/list', response_model=TemplateListResponse)
async def list_templates():
    """获取所有可用模板列表"""
    engine = get_engine()
    templates = engine.list_templates()
    return TemplateListResponse(templates=templates)


@router.post('/generate')
async def generate_from_template(req: TemplateGenerateRequest):
    """
    从题目文本生成模板讲解动画。
    
    流程：
    1. Layer1 分析题目
    2. 匹配模板（或使用指定的）
    3. LLM 提取参数
    4. 渲染模板
    5. 输出 HTML
    """
    try:
        engine = get_engine()

        # ---- 步骤1: 获取题目分析 ----
        # 调用现有 Layer1 的分析能力
        try:
            from layer1_agent import analyze_problem_safe
            analysis = analyze_problem_safe(req.text)
        except ImportError:
            # Layer1 不可用时，使用最小分析
            analysis = {
                'topic': '',
                'tags': [],
                'concepts': [],
                'text': req.text
            }

        # ---- 步骤2: 匹配模板 ----
        if req.template_id:
            tid = req.template_id
            # 校验模板是否存在
            valid_ids = [t['id'] for t in engine.list_templates()]
            if tid not in valid_ids:
                return TemplateGenerateResponse(
                    success=False,
                    fallback=False,
                    message=f'指定的模板不存在: {tid}，可用: {valid_ids}'
                )
        else:
            tid = engine.match_template(analysis)

        if not tid:
            # 无匹配 → 降级到旧流水线
            logger.info(f'无匹配模板，降级到旧流水线')
            return TemplateGenerateResponse(
                success=False,
                fallback=True,
                message='未匹配到合适的动画模板，将使用标准讲解流水线',
                template_id=None
            )

        # ---- 步骤3: LLM 提取参数 ----
        try:
            params = engine.extract_params(req.text, tid)
        except Exception as e:
            logger.error(f'LLM 参数提取失败: {e}')
            # LLM 失败时使用默认参数
            params = {}
            for key, info in engine.registry[tid]['param_schema'].items():
                params[key] = info['default']

        # ---- 步骤4: 渲染模板 ----
        html = engine.render(tid, params)

        # ---- 步骤5: 写入文件（供调试/分享） ----
        output_path = 'frontend/premium-output.html'
        try:
            with open(output_path, 'w', encoding='utf-8') as f:
                f.write(html)
        except Exception as e:
            logger.warning(f'写入输出文件失败: {e}')

        return TemplateGenerateResponse(
            success=True,
            template_id=tid,
            params=params,
            html=html,
            fallback=False,
            message='生成成功'
        )

    except Exception as e:
        logger.exception('模板生成异常')
        raise HTTPException(status_code=500, detail=str(e))


@router.post('/preview', response_class=HTMLResponse)
async def preview_template(req: TemplatePreviewRequest):
    """

    """

## 4. 错误处理策略

### 4.1 降级流程

```
generate_from_template()
  │
  ├── Layer1 分析失败 → 使用空分析 {}，继续
  │
  ├── 无匹配模板 → 返回 fallback=true，前端跳转旧流水线
  │
  ├── LLM 提取失败 → 使用模板参数默认值，标记 warning
  │
  ├── 渲染失败 → 返回 500 错误
  │
  └── 成功 → 返回 HTML
```

### 4.2 LLM 调用失败处理

```python
def _call_llm_with_fallback(self, prompt: str) -> dict:
    """带降级的 LLM 调用"""
    try:
        response = self._call_llm(prompt)
        params = self._parse_llm_json(response)
        if params:
            return params
    except Exception as e:
        logger.warning(f'LLM 调用失败，使用默认参数: {e}')

    # 降级：返回空字典，由 _validate_and_fill 填充默认值
    return {}
```

### 4.3 超时处理

LLM 调用设置 30 秒超时。如果超过 30 秒无响应，使用默认参数继续。

## 9. 附录：关键正则模式

### 9.1 LLM JSON 响应清理

```python
# 从 LLM 响应中提取纯净 JSON
JSON_PATTERN = re.compile(r'\{[^{}]*(\{[^{}]*\}[^{}]*)*\}', re.DOTALL)

def extract_json_from_response(response: str) -> str:
    """
    从 LLM 响应中提取 JSON 字符串。
    处理：
    - 前后 markdown 代码块 ```json ... ```
    - 前后空白
    - 多余文字前缀/后缀
    """
    text = response.strip()

    # 去掉 markdown 代码块标记
    text = re.sub(r'^```(?:json)?\s*', '', text)
    text = re.sub(r'\s*```$', '', text)

    # 从第一个 { 到最后一个 }
    start = text.find('{')
    end = text.rfind('}')
    if start >= 0 and end > start:
        return text[start:end + 1]

    return text
```

### 9.2 参数单位清理

```python
# 清除参数值中的单位字符
UNIT_PATTERN = re.compile(r'^([\d.eE+\-]+)\s*[a-zA-Z°%/]+$')

def clean_param_value(value: str) -> str:
    """去掉数值末尾的单位"""
    m = UNIT_PATTERN.match(value.strip())
    if m:
        return m.group(1)
    return value
```
